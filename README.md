import time
import RPi.GPIO as GPIO

GPIO.setmode(GPIO.BCM)

GPIO.setup(4, GPIO.OUT)

pin = 4
interval = 0.1

dot = .2
dash = .6

text = 'sos-sos-sos-sos'

def s ():
    GPIO.output(4, GPIO.HIGH)
    time.sleep(dot)
    GPIO.output(4, GPIO.LOW)
    time.sleep(interval)
    GPIO.output(4, GPIO.HIGH)
    time.sleep(dot)
    GPIO.output(4, GPIO.LOW)
    time.sleep(interval)
    GPIO.output(4, GPIO.HIGH)
    time.sleep(dot)
    GPIO.output(4, GPIO.LOW)
    time.sleep(interval)

    time.sleep(interval)

for c in text:
